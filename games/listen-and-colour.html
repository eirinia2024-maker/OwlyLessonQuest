function playAudio(src) { stopAudio(); currentAudio = new Audio(src); currentAudio.play().catch(() => { setStatus("Audio could not play.", "error"); }); } function setStatus(text, type = "") { statusBar.textContent = text; statusBar.className = "status-bar"; if (type) statusBar.classList.add(type); } function updatePaletteActive() { document.querySelectorAll(".color-btn").forEach(btn => { btn.classList.toggle("active", btn.dataset.key === currentChoice); }); } function removeColorFromPalette(key) { const btn = document.querySelector(`.color-btn[data-key="${key}"]`); if (!btn) return; btn.classList.add("used"); setTimeout(() => { btn.remove(); }, 220); } function chooseColor(key) { if (colored.has(key)) return; currentChoice = key; updatePaletteActive(); const item = itemMap[key]; playAudio(item.audio); setStatus(`Listen and colour the ${item.label}.`); } function showLayer(key) { const layer = document.getElementById(`layer-${key}`); if (layer) layer.classList.add("visible"); } function showFinishOverlay() { finishOverlay.classList.add("show"); } function checkComplete() { if (colored.size === items.length) { setStatus("Well Done!", "success"); setTimeout(showFinishOverlay, 350); } } items.forEach(item => { const button = document.createElement("button"); button.className = "color-btn"; button.type = "button"; button.dataset.key = item.key; button.innerHTML = `
${item.colorName}
`; button.addEventListener("click", () => chooseColor(item.key)); palette.appendChild(button); }); hotspots.forEach(hotspot => { hotspot.addEventListener("click", () => { const clickedObject = hotspot.dataset.object; if (!currentChoice) { setStatus("Tap a colour first.", "error"); return; } const target = itemMap[currentChoice]; if (clickedObject === currentChoice) { if (!colored.has(currentChoice)) { colored.add(currentChoice); showLayer(currentChoice); removeColorFromPalette(currentChoice); setStatus(`Great! You coloured the ${target.label}.`, "success"); currentChoice = null; updatePaletteActive(); checkComplete(); } } else { setStatus("Try again.", "error"); } }); }); finishBtn.addEventListener("click", () => { finishOverlay.classList.remove("show"); if (window.parent && window.parent !== window) { window.parent.postMessage("coloring_completed", "*"); } });
